                      // This program draws a target that follows the mouse
                      var line;
                      var line2;
                      function start(){
                          var x=0;
                          var y=(getHeight()/2);
                          var x1=getWidth();
                          var M = getWidth()/2;
                          var N=0;
                          var y1=getHeight();
                          line=new Line(x,y,x1,y);
                          line2=new Line(M,N,M,y1);
                          add(line2);
                          add(line);
                          mouseMoveMethod(cross);
                          mouseClickMethod(dot);
                      }
                      function dot(e){
                          var circle=new Circle(10);
                          circle.setPosition(e.getX(), e.getY());
                          circle.setColor(Color.red);
                          add(circle);
                      }
                      function cross(e){
                          line.setPosition(e.getX(),0);
                          line.setEndpoint(e.getX(),getHeight());
                          line2.setPosition(0,e.getY());
                          line2.setEndpoint(getWidth(),e.getY());
                      }
